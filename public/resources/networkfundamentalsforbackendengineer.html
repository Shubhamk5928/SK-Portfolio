Socket programming provides a low-level interface for building network applications [714].

TCP Server

import socket
import threading

def handle_client(client_socket):
    request = client_socket.recv(1024)
    response = f"Received: {request.decode()}"
    client_socket.send(response.encode())
    client_socket.close()

server = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
server.bind(('localhost', 8080))
server.listen(5)
print("TCP Server listening on port 8080")

while True:
    client_socket, addr = server.accept()
    print(f"Connection from {addr}")
    threading.Thread(target=handle_client, 
                    args=(client_socket,)).start()

Practical Python: HTTP Client

Using the requests library for HTTP communication [475]. 

import requests
from requests.exceptions import RequestException

def make_http_request(url, method='GET', data=None, headers=None):
    """
    Make an HTTP request with error handling and response processing
    """
    try:
        if method.upper() == 'GET':
            response = requests.get(url, headers=headers, timeout=10)
        elif method.upper() == 'POST':
            response = requests.post(url, json=data, headers=headers, timeout=10)
        else:
            raise ValueError(f"Unsupported HTTP method: {method}")
        
        # Check response status
        response.raise_for_status()
        
        # Process successful response
        return {
            'status_code': response.status_code,
            'headers': dict(response.headers),
            'body': response.json() if response.headers.get('content-type', '').startswith('application/json') else response.text,
            'elapsed_ms': response.elapsed.total_seconds() * 1000
        }
        
    except RequestException as e:
        print(f"HTTP request failed: {e}")
        return None
    except Exception as e:
        print(f"Unexpected error: {e}")
        return None

# Usage example
if __name__ == "__main__":
    # GET request
    result = make_http_request('https://api.example.com/users')
    if result:
        print(f"Status: {result['status_code']}")
        print(f"Response time: {result['elapsed_ms']}ms")
        
    # POST request with JSON
    user_data = {'name': 'John Doe', 'email': 'john@example.com'}
    result = make_http_request('https://api.example.com/users', 'POST', user_data)
    if result:
        print(f"Created user with ID: {result['body'].get('id')}")

import socket
import struct
import time
import os

def create_icmp_packet(seq_number=1):
    """Create ICMP echo request packet"""
    # ICMP header: type (8), code (0), checksum, id, sequence
    icmp_type = 8  # Echo Request
    icmp_code = 0
    icmp_id = os.getpid() & 0xFFFF
    icmp_seq = seq_number
    icmp_checksum = 0
    
    # Create header with checksum 0
    header = struct.pack('BBHHH', icmp_type, icmp_code, 
                        icmp_checksum, icmp_id, icmp_seq)
    
    # Calculate checksum
    data = b'PingData'  # Payload
    packet = header + data
    checksum = calculate_checksum(packet)
    
    # Create final packet with correct checksum
    header = struct.pack('BBHHH', icmp_type, icmp_code, 
                        checksum, icmp_id, icmp_seq)
    return header + data

def ping(host, timeout=2):
    """Send ICMP echo request and measure response time"""
    try:
        # Resolve hostname to IP
        ip_address = socket.gethostbyname(host)
        print(f"Pinging {host} ({ip_address})")
        
        # Create raw socket
        icmp_socket = socket.socket(socket.AF_INET, 
                                   socket.SOCK_RAW, 
                                   socket.IPPROTO_ICMP)
        
        # Set timeout
        icmp_socket.settimeout(timeout)
        
        # Create and send packet
        packet = create_icmp_packet(seq_number=1)
        start_time = time.time()
        
        icmp_socket.sendto(packet, (ip_address, 1))
        
        # Receive response
        response, addr = icmp_socket.recvfrom(1024)
        end_time = time.time()
        
        # Calculate round-trip time
        rtt_ms = (end_time - start_time) * 1000
        
        print(f"Reply from {addr[0]}: time={rtt_ms:.2f}ms")
        return rtt_ms
        
    except socket.timeout:
        print(f"Request timed out for {host}")
        return None
    except socket.gaierror:
        print(f"Could not resolve hostname: {host}")
        return None
    finally:
        icmp_socket.close()

Python Load Balancer Simulation

import socket
import threading
import random
from typing import List, Tuple

class LoadBalancer:
    def __init__(self, host: str, port: int, backend_servers: List[Tuple[str, int]]):
        self.host = host
        self.port = port
        self.backend_servers = backend_servers
        self.current_index = 0
        self.lock = threading.Lock()
    
    def get_next_server_round_robin(self) -> Tuple[str, int]:
        """Round-robin server selection"""
        with self.lock:
            server = self.backend_servers[self.current_index]
            self.current_index = (self.current_index + 1) % len(self.backend_servers)
            return server
    
    def get_server_least_connections(self, connections: dict) -> Tuple[str, int]:
        """Select server with least connections"""
        with self.lock:
            min_connections = min(connections.values())
            candidates = [s for s, c in connections.items() if c == min_connections]
            return random.choice(candidates)
    
    def handle_client(self, client_socket: socket.socket, 
                     server_socket: socket.socket):
        """Handle bidirectional communication"""
        try:
            while True:
                # Forward client to server
                data = client_socket.recv(4096)
                if not data:
                    break
                server_socket.send(data)
                
                # Forward server to client
                data = server_socket.recv(4096)
                if not data:
                    break
                client_socket.send(data)
        except socket.error as e:
            print(f"Connection error: {e}")
        finally:
            client_socket.close()
            server_socket.close()
    
    def start(self, algorithm="round_robin"):
        """Start the load balancer"""
        server = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        server.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1)
        server.bind((self.host, self.port))
        server.listen(5)
        
        print(f"Load balancer started on {self.host}:{self.port}")
        print(f"Backend servers: {self.backend_servers}")
        print(f"Algorithm: {algorithm}")
        
        try:
            while True:
                client_socket, client_addr = server.accept()
                print(f"Connection from {client_addr}")
                
                # Select backend server
                if algorithm == "round_robin":
                    server_addr = self.get_next_server_round_robin()
                else:
                    # For demonstration, use round-robin
                    server_addr = self.get_next_server_round_robin()
                
                # Connect to backend server
                server_socket = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
                server_socket.connect(server_addr)
                
                # Start forwarding threads
                threading.Thread(target=self.handle_client, 
                               args=(client_socket, server_socket)).start()
                
        except KeyboardInterrupt:
            print("\nShutting down load balancer...")
        finally:
            server.close()

# Usage example
if __name__ == "__main__":
    # Define backend servers
    backend_servers = [
        ('localhost', 8001),
        ('localhost', 8002),
        ('localhost', 8003)
    ]
    
    # Create and start load balancer
    lb = LoadBalancer('localhost', 8080, backend_servers)
    lb.start(algorithm="round_robin")

Interview Questions for Backend Engineers

Common networking questions asked in backend engineering interviews

Core Networking Concepts

1. Explain the difference between TCP and UDP.

TCP (Transmission Control Protocol)
  • • Connection-oriented protocol
  • • Reliable, ordered delivery
  • • Error checking and recovery
  • • Flow and congestion control
  • • Three-way handshake
  • • Suitable for: HTTP, FTP, SMTP
UDP (User Datagram Protocol)
  • • Connectionless protocol
  • • Unreliable delivery
  • • No error recovery
  • • Minimal overhead
  • • No handshake required
  • • Suitable for: DNS, VoIP, gaming

2. What happens when you type a URL in your browser?

  1. URL Parsing: Browser extracts domain name from URL
  2. DNS Resolution:
    • • Browser cache → OS cache → Recursive resolver
    • • Root server → TLD server → Authoritative server
  3. TCP Connection: Three-way handshake with server
  4. TLS Handshake: (If HTTPS) Certificate validation and key exchange
  5. HTTP Request: Browser sends HTTP request headers
  6. Server Processing: Application processes request
  7. HTTP Response: Server sends response with status and body
  8. Page Rendering: Browser parses HTML, CSS, JavaScript

3. Describe the OSI or TCP/IP model.

TCP/IP Model (4 layers): Practical implementation used in modern networking

Application Layer

HTTP, FTP, SMTP, DNS - Direct user interaction

Transport Layer

TCP, UDP - End-to-end communication

Internet Layer

IP, ICMP - Routing and addressing

Network Access Layer

Ethernet, Wi-Fi - Physical transmission

Deep Dive into TCP

1. Walk me through the TCP three-way handshake.

1
Client → Server: SYN

Client sends SYN (synchronize) packet with initial sequence number X

2
Server → Client: SYN-ACK

Server responds with SYN (sequence Y) and ACK (X+1)

3
Client → Server: ACK

Client sends ACK (Y+1), connection established

2. What is the purpose of the TIME_WAIT state?

Duration: Typically 2 × Maximum Segment Lifetime (MSL), usually 2-4 minutes

Purpose 1: Ensures the final ACK is received by the passive closer

Purpose 2: Prevents old, duplicate packets from being accepted by new connections using the same 4-tuple (source/destination IP and ports)

Impact: Can cause port exhaustion on high-traffic servers; mitigation includes connection pooling and proper server configuration

3. How does TCP handle packet loss and ensure reliable delivery?

Mechanisms
  • Sequence Numbers: Each byte has a sequence number for ordered reassembly
  • Acknowledgments: Receiver sends ACK for received packets
  • Retransmission: Sender retransmits unacknowledged packets
  • Timeout Calculation: Dynamic timeout based on RTT measurements
Process
  1. Sender transmits packet with sequence number
  2. Receiver sends ACK with next expected sequence number
  3. If ACK not received within timeout, sender retransmits
  4. Receiver discards duplicate packets but still ACKs them
  5. Fast retransmit triggers after 3 duplicate ACKs

HTTP and Application Layer

1. What are the key differences between HTTP/1.1 and HTTP/2?

HTTP/1.1 Limitations
  • • Head-of-line blocking
  • • Multiple TCP connections needed
  • • Text-based protocol overhead
  • • No header compression
  • • Client must request all resources
HTTP/2 Improvements
  • • Multiplexing (multiple streams per connection)
  • • HPACK header compression
  • • Server push capability
  • • Binary protocol for efficiency
  • • Stream prioritization

2. What is HTTP Keep-Alive and why is it important?

Definition: HTTP Keep-Alive (persistent connections) allows multiple HTTP requests/responses to be sent over a single TCP connection

Benefits:

  • • Reduces connection establishment overhead (TCP handshake)
  • • Lower latency for subsequent requests
  • • Better resource utilization
  • • Improved performance for web pages with multiple resources

Configuration: Controlled by Connection: keep-alive header and timeout parameters

3. How does DNS work, and what are some common record types?

DNS Resolution Process
  1. Browser checks local cache
  2. Query recursive resolver (ISP)
  3. Query root name servers
  4. Query TLD name servers
  5. Query authoritative name servers
  6. Cache result and return IP
Common Record Types
A: IPv4 address mapping
AAAA: IPv6 address mapping
CNAME: Canonical name (alias)
MX: Mail exchange servers
TXT: Text records (SPF, DKIM)
NS: Name server delegation

Load Balancing and Scaling

1. What is the difference between a Layer 4 and a Layer 7 load balancer?

Layer 4 (Transport)
  • Operates at: Transport layer (TCP/UDP)
  • Decisions based on: IP addresses and ports
  • Speed: Very fast (no content inspection)
  • Features: Basic load distribution
  • Use cases: TCP/UDP traffic, databases
Layer 7 (Application)
  • Operates at: Application layer (HTTP)
  • Decisions based on: HTTP headers, URL, cookies
  • Speed: Slower (content inspection)
  • Features: SSL termination, content-based routing
  • Use cases: Web applications, API gateways

2. How does Round-Robin DNS work as a load balancing strategy?

Process:

  1. Multiple A/AAAA records created for same domain
  2. Each record points to different server IP
  3. DNS server rotates order of IP addresses in responses
  4. Clients typically use first IP in the list

Advantages:

  • Simple to implement
  • No additional hardware required
  • Works with any DNS provider

Limitations:

  • No health checking (sends traffic to failed servers)
  • No session persistence
  • TTL caching can delay failover
  • Can't account for server load or capacity

3. What are some challenges of managing state in a load-balanced environment?

Challenges
  • • Session persistence (sticky sessions) required
  • • Load distribution becomes uneven
  • • Server failure leads to session loss
  • • Scaling issues with session storage
  • • Complexity in session replication
Solutions
  • Sticky Sessions: Load balancer routes same client to same server
  • Session Replication: Copy session data across all servers
  • Centralized Session Store: Redis, Memcached, database
  • JWT Tokens: Stateless authentication with tokens
  • Database Sessions: Store session data in database
